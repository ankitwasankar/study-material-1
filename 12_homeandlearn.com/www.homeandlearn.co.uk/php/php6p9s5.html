Loop round values in a PHP array

 

<?PHP

$start = 1;
$times = 2;
$answer = array();

for ($start; $start < 11; $start++) {

$answer[$start] = $start * $times;

}

print $answer[1] . " ";
print $answer[4] . " ";
print $answer[8] . " ";
print $answer[10];

?>
