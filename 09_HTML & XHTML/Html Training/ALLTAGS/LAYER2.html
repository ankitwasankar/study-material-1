<SCRIPT>
var nMove0 = -10;
var nMove1 =   5;

function ChangeLayer(nLayer) {
var xLayer = document.layers["Layer" + nLayer]

  if (xLayer.visibility == "hide") {
     xLayer.visibility = "show"}
  else {
     xLayer.visibility = "hide"}
}

function moveLayer() {
var xLayer ;

  xLayer = document.layers["Layer1"];
  if (xLayer.left < 10 || xLayer.left > 500){
     nMove0 *= -1}
  xLayer.left += nMove0;

  xLayer = document.layers["Layer2"];
  if (xLayer.top < 10 || xLayer.top > 400){
     nMove1 *= -1}
  xLayer.top += nMove1;

  setTimeout("moveLayer()", 10);
  return;
}
</SCRIPT>

<FORM>
